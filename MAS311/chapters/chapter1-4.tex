\section{Generators of Groups and Free Groups}

\thm[]{}{
    Let \(G\) be a group and \(\OO \subsetneq S \subseteq G\).
    Let \(\lang S \rang\) be the closure of \(S\) under
    the structure \((G, \cdot, {}\inv)\).
    \begin{enumerate}[nolistsep, label=(\roman*), ref=\protect{(\roman*)}, listparindent=\parindent]
        \ii \(\lang S \rang \le G\) and \(S \subseteq \lang S \rang\).
        \ii If \(H \le G\) and \(S \subseteq H\), then \(\lang S \rang \subseteq H\).
    \end{enumerate}
}
\pf{Proof}{
    Trivial.
}

\dfn[generator]{Generator}{
    Let \(G\) be a group and \(\OO \subsetneq S \subseteq G\).
    If \(G = \lang S \rang\), then we say
    \(G\) is \textit{generated by \(S\)} and \(S\) is a \textit{generator} of \(G\).
    If \(S\) is finite, then \(G\) is \textit{finitely generated}.
}

\exmp[finGenExmp]{}{
    \begin{enumerate}[nolistsep, label=(\roman*), ref=\protect{\Cref{exmp:finGenExmp} (\roman*)}, listparindent=\parindent]
        \ii
        A finite group is finitely generated. \(G = \lang G \rang\).
        \ii
        \(\ZZ = \lang-1\rang\) is finitely generated.
        \ii\label{itm:finGenExmp.iii}
        \(\QQ\) is not finitely generated.
        If \(\QQ = \lang\,p_i/q_i \mid i < n\,\rang\),
        then, for a prime \(p \in \PP\) such that \(\fall i < n,\: p \nmid q_i\),
        we have \(1/p \notin \lang\,p_i/q_i \mid i < n\,\rang\).
        \ii
        \(D_n = \lang r_1, s \rang\). (This is a minimal representation.)
        \ii
        \(Q_8 = \lang i, j \rang = \lang j, k \rang = \lang k, i \rang\).
    \end{enumerate}
}

\dfn[]{Group Presentation}{
    We write
    \[
        G = \lang\,S \mid R\,\rang
    \]
    as a way of representing group \(G\) in terms of
    \textit{generator} \(S\) and a set of relations \(R\).
}

\exmp[]{}{
\begin{enumerate}[nolistsep, label=(\roman*), ref=\protect{(\roman*)}, listparindent=\parindent]
    \ii \(\ZZ = \lang 1 \rang\).
    \ii \(D_n = \lang\,r, s \mid r^n = s^2 = rsrs = 1\,\rang\).
\end{enumerate}
}

\thm[]{}{
    Let \(G = \lang\,g_1, \cdots, g_k \mid r_1(g_1, \cdots, g_k) = \cdots = r_m(g_1, \cdots, g_k) = 1\,\rang\)
    be a group presentation.
    Let \(H\) be a group.
    If \(\vphi \colon \{\,g_1, \cdots, g_k\,\} \to H\)
    such that \(r_i(\vphi(g_1), \cdots, \vphi(g_k)) = 1\) for all \(i \in [m]\),
    then there uniquely exists a group homomorphism
    \(\tilde \vphi \colon G \to H\) such that
    \(\restr{\tilde \vphi}{\{g_1,\cdots,g_k\}} = \vphi\).
}
