\subsection*{Selected Problems}

\exer[2.1.1]{}{
    If \(a, b \in A\), then \((a, b) \in \mcal P(\mcal P(A))\).
}
\pf{Proof}{
    If \(a, b \in A\), then \(\{a\}, \{a, b\} \in \mcal P(A)\),
    and thus \((a, b) = \{\{a\}, \{a, b\}\} \subseteq \mcal P(A)\).
}
